## 6. 二叉树的最近公共祖先

给定一个二叉树, 找到该树中两个指定节点的最近公共祖先。

```javascript
var lowestCommonAncestor = function(root, p, q) {
    // 决定性一击！
    if (!root || root === p || root === q) {
        return root;
    }
    let left = lowestCommonAncestor(root.left, p, q);
    let right = lowestCommonAncestor(root.right, p, q);

    // 都存在证明都找到了，返回父节点
    if (left && right) {
        return root;
    }

    // 只有一个存在，说明是祖先节点
    return left || right;
};
```

var isValidBST = function(root) {
    const isBSTHelper = (node, lower, upper) => {
        if (node === null) {
            return true;
        }

        if (lower >= node.val || upper <= node.val) {
            return false;
        }

        return isBSTHelper(node.left, lower, node.val) && isBSTHelper(node.right, node.val, upper);
    };

};

## 9. 对称二叉树

判断二叉树是否对称

```javascript
var isSymmetric = function (root) {
    if (!root) {
        return true;
    }
    const isMirror = (left, right) => {
        if (!left && !right) {
            return true;
        }
        if (!left || !right) {
            return false;
        }
        if (left.val !== right.val) {
            return false;
        }
        return isMirror(left.left, right.right) && isMirror(left.right, right.left);
    };
    return isMirror(root.left, root.right);
};
```

## 124. 二叉树中的最大路径和

路径和 是路径中各节点值的总和。

给你一个二叉树的根节点 root ，返回其 最大路径和 。

输入：root = [1,2,3]
输出：6
解释：最优路径是 2 -> 1 -> 3 ，路径和为 2 + 1 + 3 = 6

```javascript
/**
 * @param {TreeNode} root
 * @return {number}
 */
var maxPathSum = function(root) {
    let maxSum = 0;
    const maxPathSumHelper = (root) => {
        if (!root) {
            return 0;
        }

        const leftMax = Math.max(maxPathSumHelper(root.left), 0);
        const rightMax = Math.max(maxPathSumHelper(root.right), 0);

        const currentPathSum = root.val + leftMax + rightMax;
        maxSum = Math.max(maxSum, currentPathSum);

        return root.val + Math.max(leftMax, rightMax);
    };

    maxPathSumHelper(root);
    return maxSum;
};
```
