## 1. 快速排序

```javascript
const quickSort = (arr, left, right) => {
    const partition = (arr, left, right) => {
        const pivot = arr[right];
        let i = left;

        for (let j = left; j < right; j++) {
            if (arr[j] <= pivot) {
                // 将小于基准元素的数字放到左边
                [arr[i], arr[j]] = [arr[j], arr[i]];
                i++;
            }
        }

        // 将基准元素放到中间
        [arr[i], arr[right]] = [arr[right], arr[i]];
        return i;
    };
    if (left < right) {
        const pivotIndex = partition(arr, left, right);
        quickSort(arr, left, pivotIndex - 1);
        quickSort(arr, pivotIndex + 1, right);
    }
};
```

```javascript
var compareVersion = function(version1, version2) {
    const v1 = version1.split('.');
    const v2 = version2.split('.');

    for (let i = 0; i < v1.length || i < v2.length; i++) {
        let x = 0, y = 0;
```

## 3. 节流 & 防抖
 
节流：规定一个单位时间内，只能触发一次事件。如果这个单位时间内触发多次事件，只有一次生效。

防抖：规定一个单位时间内，只能触发一次事件。如果在这个单位时间内连续触发多次事件，只有最后一次生效。

```javascript
function throttled(fn, delay) {
    let timer = null;

    return function() {
        if (timer) {
            return;
        }
        timer = setTimeout(() => {
            fn.apply(this, arguments);
            timer = null;
        }, delay);
    }
}

function debounce(fn, delay) {
    let timer = null;

    return function() {
        clearTimeout(timer);
        timer = setTimeout(() => {
            fn.apply(this, arguments);
        }, delay);
    }
}

// 最后一次必须执行的节流函数
function throttle(fn, delay) {
    let timer = null;
    let lastTime = 0;
    let lock = false;
    
    return function () {
        const now = Date.now();
        if (!lastTime || now - lastTime >= delay) {
            clearTimeout(timer);
            fn.apply(this, arguments);
            lastTime = now;
            lock = false;
        }
        else {
            if (!lock) {
                lock = true;
                timer = setTimeout(() => {
                    if (lock) {
                        fn.apply(this, arguments);
                        lastTime = Date.now();
                        lock = false;
                    }
                }, delay);
            }
        }
    }
}
```

```javascript
function promiseAll(promises) {
    return new Promise((resolve, reject) => {
        let count = 0;
        let result = [];

        if (!Array.isArray(promises)) {
            return reject(new Error('must be an array'));
        }

        const handleComplete = (index, value) => {
            count++;
            result[index] = value;
            if (count === promises.length) {
                resolve(result);
            }
        };

        for (let i = 0; i < promises.length; i++) {
            let current = promises[i];
            if (typeof current.then === 'function') {
                current.then(value => handleComplete(i, value)).catch(err => reject(err));
            }
            else {
                handleComplete(i, current);
            }
        }
    });
}

function promiseRace(promises) {
    return new Promise((resolve, reject) => {
        promises.forEach(promise => {
            promise.then(resolve).catch(reject);
        });
    });
}
```

```javascript
class MyPromise {
    constructor(fn) {
        this.status = 'pending';
        this.value = null;
        this.reason = null;
        this.onResolvedCallbacks = [];
        this.onRejectedCallbacks = [];

        const resolve = value => {
            if (this.status === 'pending') {
                this.status = 'fulfilled';
                this.value = value;
                this.onResolvedCallbacks.forEach(fn => fn());
            }
        };

        const reject = reason => {
            if (this.status === 'pending') {
                this.status = 'rejected';
                this.reason = reason;
                this.onRejectedCallbacks.forEach(fn => fn());
            }
        };
        try {
            fn(resolve, reject);
        }
        catch (err) {
            reject(err);
        }
    }

    then(onFulfilled, onRejected) {
        onFulfilled = typeof onFulfilled === 'function' ? onFulfilled : value => value;
        onRejected = typeof onRejected === 'function' ? onRejected : reason => { throw reason };

        const newPromise = new MyPromise((resolve, reject) => {
            if (this.status === 'fulfilled') {
                onFulfilled(this.value);
            }
            else if (this.status === 'rejected') {
                onRejected(this.reason);
            }
            else {
                this.onResolvedCallbacks.push(onFulfilled);
                this.onRejectedCallbacks.push(onRejected);
            }
        });

        return newPromise;
    }

    catch(onRejected) {
        return this.then(null, onRejected);
    }

    finally(onFinally) {
        return this.then(
            value => MyPromise.resolve(onFinally()).then(() => value),
            reason => MyPromise.resolve(onFinally()).then(() => { throw reason })
        );
    }

    static resolve(value) {
        return new MyPromise(resolve => resolve(value));
    }

    static reject(reason) {
        return new MyPromise((resolve, reject) => reject(reason));
    }

    static race(promises) {
        return new MyPromise((resolve, reject) => {
            promises.forEach(promise => {
                MyPromise.resolve(promise).then(
                    value => resolve(value),
                    reason => reject(reason)
                );
            });
        });
    }

    allSettled(promises) {
        return Promise.all(promises.map(promise =>
            Promise.resolve(promise)
            .then(value => ({ status: 'fulfilled', value }))
            .catch(reason => ({ status: 'rejected', reason }))
        ));
    }
}
```

```javascript
/**
 * @param {number} capacity
 */
var LRUCache = function(capacity) {
    this.capacity = capacity;
    this.cache = new Map();
    this.order = new Map();
};

/** 
 * @param {number} key
 * @return {number}
 */
LRUCache.prototype.get = function(key) {
    if (this.cache.has(key)) {
        this.order.delete(key);
        this.order.set(key, true);
        return this.cache.get(key);
    }
    else {
        return -1;
    }
};

/** 
 * @param {number} key 
 * @param {number} value
 * @return {void}
 */
LRUCache.prototype.put = function(key, value) {
    if (this.cache.has(key)) {
        this.cache.set(key, value);
        this.order.delete(key);
        this.order.set(key, true);
    }
    else {
        if (this.cache.size >= this.capacity) {
            const oldestKey = this.order.keys().next().value;
            this.cache.delete(oldestKey);
            this.order.delete(oldestKey);
        }
        this.cache.set(key, value);
        this.order.set(key, true);
    }
};

/**
 * Your LRUCache object will be instantiated and called as such:
 * var obj = new LRUCache(capacity)
 * var param_1 = obj.get(key)
 * obj.put(key,value)
 */
```

## 并发任务队列 - Promise

```javascript
class ConcurrencyQueue {
    constructor(maxConcurrency) {
        this.maxConcurrency = maxConcurrency;
        this.currentConcurrency = 0;
        this.queue = [];
    }

    enqueue(task) {
        return new Promise((resolve, reject) => {
            this.queue.push({
                task,
                resolve,
                reject
            });
            this.processQueue();
        });
    }

    async processQueue() {
        if (this.currentConcurrency < this.maxConcurrency && this.queue.length) {
            const { task, resolve, reject } = this.queue.shift();
            this.currentConcurrency++;
            try {
                const result = await task();
                resolve(result);
            }
            catch (error) {
                reject(error);
            }
            finally {
                this.currentConcurrency--;
                this.processQueue();
            }
        }
    }
}

// 示例使用
```

```javascript
function myNew(constructor, ...args) {
    // 步骤 1：创建一个空的对象
    const obj = {};

    // 步骤 2：将新对象的原型链接到构造函数的原型对象
    Object.setPrototypeOf(obj, constructor.prototype);

    // 步骤 3：将构造函数的上下文（this）绑定到新创建的对象
    const result = constructor.apply(obj, args);

    // 步骤 4：如果构造函数返回了一个对象，则返回该对象；否则，返回新创建的对象
    return result instanceof Object ? result : obj;
}
```

```javascript
function deepCopy(obj) {
    // 检查是否为基本数据类型，如果是，则直接返回
    if (obj === null || typeof obj !== 'object') {
        return obj;
    }

    // 根据对象的类型创建一个新的对象或数组
    const copy = Array.isArray(obj) ? [] : {};

    // 递归地复制对象的每个属性或数组元素
    for (let key in obj) {
        if (obj.hasOwnProperty(key)) {
            copy[key] = deepCopy(obj[key]);
        }
    }

    return copy;
}
```

## 发布订阅模式

```javascript
class EventEmit {
    constructor() {
        this.events = {};
    }

    // 监听
    on(eventName, listener) {
        if (!this.events[eventName]) {
            this.events[eventName] = [];
        }
        this.events[eventName].push(listener);
    }

    // 发布
    emit(eventName, ...args) {
        const listeners = this.events[eventName];
        if (listeners) {
            listeners.forEach(listener => listener(...args));
        }
    }

    // 关闭监听
    off(eventName, listener) {
        if (this.events[eventName]) {
            this.events[eventName] = this.events[eventName].filter(item => item !== listener);
        }
    }

    // 监听一次
    once(eventName, listener) {
        const onceListener = (...args) => {
            listener(...args);
            this.off(eventName, onceListener);
        };
        this.on(eventName, onceListener);
    }
}
```

```javascript
function myInstanceof(left, right) {
    // 这里先用 typeof 来判断基础数据类型，如果是，直接返回 false
    if (typeof left !== 'object' || left === null) {
        return false;
    }
    // getProtypeOf 是 Object 对象自带的API，能够拿到参数的原型对象
    let proto = Object.getPrototypeOf(left);
    while (true) {                  
        if (proto === null) {
            return false;
        }
        if (proto === right.prototype) {
            return true; // 找到相同原型对象，返回true
        }
        proto = Object.getPrototypeOf(proto);

}
```

```javascript
    const flatten = (array, depth) => {
        for (const item of array) {
            if (Array.isArray(item) && depth > 0) {
```

```javascript
function curry(func) {
    return function curried(...args) {
        if (args.length < func.length) {
            return function (...nextArgs) {
                return curried(...args, ...nextArgs);
            }
        }
        else {
            return func(...args);
        }
    }
}
```

```javascript
function multiRequest(urls = [], maxNum) {
    let len = urls.length;
    let result = new Array(len).fill(false);
    let count = 0;

    return new Promise((resolve, reject) => {
        function next() {
            let current = count++;
            if (current >= len) {
                if (!result.includes(false)) {
                    resolve(result);
                }
                return;
            }

                console.log('结束: ' + current);

                if (current < len) {
                    next();
                }
            });
        }

        }
    });
```
